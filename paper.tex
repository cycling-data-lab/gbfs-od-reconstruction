       $^{*}$ Author to whom any correspondence should be addressed.}

\email{rfosse@cesi.fr}

\keywords{GBFS; origin--destination reconstruction; data-standard compliance;
entropic optimal transport; shared micromobility}

\begin{abstract}
Reconstructing origin--destination (OD) flows from public bike-sharing feeds
is split between two regimes: macroscopic inference from station stock
counts, which is underdetermined, and deterministic tracking of per-vehicle
identifiers, which is exposed to temporal aliasing and to the
privacy-driven rotation of identifiers recommended by the GBFS standard.
We recast the problem as a hybrid in which a partial, identifier-tracked
micro-sample supplies the deterrence (cost) structure of an entropic
optimal-transport / gravity model whose margins are pinned by the station
counts. Two results follow. First, the transport cost is identifiable only
modulo additive station effects; the bias induced by incomplete observation
equals the \emph{non-separable} component of the log-selection probability,
so station-emptiness censoring cancels exactly, whereas 60-second
polling---whose capture probability is a function of travel time, the very
quantity being estimated---does not. Second, the estimator yields a
\emph{sampling-horizon law}: the continuous-collection time needed to
reconstruct OD to target accuracy $\delta$ scales as $\delta^{-4}$ and
inversely with the identifier-persistence rate $q$. The law implies a
standard-compliance threshold below which operational OD monitoring is
infeasible. We give a feed-by-feed audit protocol instantiating the bound
across \Todo{$N$} French networks.

\end{abstract}

\section{Data regimes, model and notation}
\label{sec:setup}

\Todo{Prose introducing the three regimes (published trips; station-status
counts; persistent-identifier tracking) and positioning our hybrid between
the latter two.}

Let stations be indexed $i,j\in\{1,\dots,N\}$ and let $\Pstar\in\R_{\ge
0}^{N\times N}$ be the true OD coupling over a fixed period, with row
margins $r^\star_i=\sum_j \Pstar_{ij}$ (outflow) and column margins
$c^\star_j=\sum_i \Pstar_{ij}$ (inflow). The structural (gravity / entropic
optimal-transport) assumption is the Gibbs form
\begin{equation}
\label{eq:gibbs}
\Pstar_{ij} = u^\star_i\, v^\star_j\, e^{-\cstar_{ij}/\varepsilon},
\end{equation}
for potentials $u^\star,v^\star$, ground cost $\cstar$ and temperature
$\varepsilon>0$. This is the maximum-entropy trip-distribution model
\citep{Wilson1967} and is identical, as an algorithm, to iterative
proportional fitting and to the Sinkhorn iteration for entropic optimal
transport \citep{Sinkhorn1964,Cuturi2013,PeyreCuturi2019}.

\paragraph{Macro observation (censored margins).}
Station-status counts provide noisy, right-censored margins
$\hat r_i = \rho_i\, r^\star_i$ with availability fraction $\rho_i\in(0,1]$,
reflecting demand lost at empty/full stations \citep{RudloffLackner2014,
Negahban2019}.

\paragraph{Micro observation (selection-tilted sample).}
A sub-sample of identifier-tracked trips is drawn from the
\emph{selection-tilted} law $\tilde P_{ij}\propto \Pstar_{ij}\,S_{ij}$,
where $S_{ij}\in[0,1]$ aggregates identifier persistence $q$, non-rebalancing
$(1-\beta)$, and capture under polling interval $\Delta$ (anti-aliasing)
$\kappa$.

\section{Identifiability of the interaction cost}
\label{sec:identif}

The Gibbs form \eqref{eq:gibbs} is invariant under
$\cstar_{ij}\mapsto \cstar_{ij}+f_i+g_j$ (absorbed into the potentials).
Hence the cost is identifiable only modulo the separable subspace
$\Ncal=\{(f_i+g_j)_{ij}\}$ ($\dim\Ncal=2N-1$); the identifiable object is
the interaction part $\Pperp \cstar$, equivalently the second-order
cross-differences $c_{ij}-c_{il}-c_{kj}+c_{kl}$.

\begin{theorem}[Cost identifiability]
\label{thm:identif}
Under \eqref{eq:gibbs}, $\Pperp\cstar$ is identifiable from a coupling, and
no separable component of $\cstar$ is. \Todo{State regularity conditions;
proof by the matching/ground-metric argument of \citet{GalichonSalanie2022,
DupuyGalichon2014,CuturiAvis2014}; full proof in SI Appendix~A.}
\end{theorem}

\begin{definition}[Admissible cost feature]
\label{def:admissible}
A candidate feature $\psi_{ij}=\phi^{(k)}(x_i,x_j)$ is \emph{admissible} iff it
survives two-way (origin/destination) centring,
$\tilde\psi_{ij}=\psi_{ij}-\bar\psi_{i\cdot}-\bar\psi_{\cdot j}+\bar\psi_{\cdot\cdot}
\neq 0$; monadic features ($\psi_{ij}=a_i$ or $b_j$) are inadmissible and
belong in the potentials/margins, not the cost.
\end{definition}

\section{The selection-bias decomposition}
\label{sec:bias}

\begin{proposition}[Bias = non-separable log-selection]
\label{prop:bias}
The micro-sample identifies the cost of $\tilde P$, namely
$\tilde c = \cstar - \varepsilon\log S$; therefore the estimated identifiable
cost satisfies
\begin{equation}
\label{eq:bias}
\Pperp\hat c \;\xrightarrow{\,n\to\infty\,}\; \Pperp\cstar
\;-\;\varepsilon\,\Pperp\log S .
\end{equation}
In particular, if the selection is separable, $S_{ij}=a_i b_j$, then
$\log S\in\Ncal$ and the bias vanishes.
\end{proposition}

\begin{corollary}[Which censoring hurts]
\label{cor:aliasing}
Station-emptiness censoring acts at the origin/destination level
($S_{ij}=a_i b_j$) and \emph{cancels} in the identifiable cost. Aliasing
under polling interval $\Delta$ has capture probability that depends on the
trip duration $t_{ij}$, hence on $\cstar_{ij}$ itself; it is
\emph{non-separable} and correlated with the estimand, so it does not cancel
and systematically attenuates the cost--distance relationship. Residual
operator rebalancing on specific corridors is likewise non-separable.
\end{corollary}

\section{The sampling-horizon law}
\label{sec:horizon}

Writing $\Lambda$ for the trip-generation rate (trips/day), the number of
clean, attributable observations after $T$ days of polling at interval
$\Delta$ is
\begin{equation}
\label{eq:neff}
\neff = \Lambda\,T\,\kappa(\Delta)\,q\,(1-\beta)\,(1-\gamma),
\end{equation}
with $\gamma$ the GPS-jitter / collision loss. Propagating the cost error
through the entropic-OT forward map (sensitivity $O(1/\varepsilon)$,
entropic bias $O(\varepsilon)$; balancing gives an $n^{-1/4}$ rate on the OD
error) yields the requirement $\neff\gtrsim \dim\theta/(\lambda_{\min}\,
\delta_\OD^{4})$ and hence the collection horizon.

\begin{theorem}[Sampling-horizon law]
\label{thm:horizon}
To reconstruct the OD coupling to relative accuracy $\delta_\OD$ under the
estimator of Sections~\ref{sec:identif}--\ref{sec:bias},
\begin{equation}
\label{eq:Tstar}
\Tstar \;=\;
\frac{\dim\theta}
     {\lambda_{\min}\,\delta_\OD^{4}\;\Lambda\,\kappa(\Delta)\,q\,(1-\beta)(1-\gamma)} ,
\end{equation}
so $\Tstar\propto \delta_\OD^{-4}\,q^{-1}\,\Lambda^{-1}\,\lambda_{\min}^{-1}$.
\Todo{State constants and the forward-stability lemma; proof in
SI Appendix~B.}
\end{theorem}

\begin{corollary}[Standard-compliance threshold]
\label{cor:threshold}
For any operational horizon $T_{\max}$ and accuracy $\delta_\OD$, OD is
reconstructible only on feeds with identifier persistence
$q \ge q_{\min}:=\dim\theta/(\lambda_{\min}\delta_\OD^4 \Lambda
\kappa(1-\beta)(1-\gamma)\,T_{\max})$. Below $q_{\min}$ the horizon exceeds
$T_{\max}$ and operational monitoring is infeasible.
\end{corollary}
